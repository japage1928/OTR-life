import { INTERNAL_LINKING_CONFIG } from "../../config/internalLinkingConfig";
import type { ClusterMatch, PostRecord } from "./types";

function normalize(text: string): string {
  return text.toLowerCase();
}

function countKeywordHits(haystack: string, keywords: string[]): number {
  let score = 0;

  for (const kw of keywords) {
    const pattern = new RegExp(
      `\\b${kw.replace(/[.*+?^${}()|[\]\\]/g, "\\$&")}\\b`,
      "gi"
    );
    const matches = haystack.match(pattern);
    if (matches) score += matches.length;
  }

  return score;
}

export function scorePostAgainstClusters(post: PostRecord): ClusterMatch[] {
  const title = normalize(post.title || "");
  const slug = normalize(post.slug || "");
  const excerpt = normalize(post.excerpt || "");
  const content = normalize(post.content || "");
  const tags = normalize(post.tags || "");

  const results: ClusterMatch[] = INTERNAL_LINKING_CONFIG.map((cluster) => {
    const titleHits = countKeywordHits(title, cluster.keywords) * 5;
    const slugHits = countKeywordHits(slug, cluster.keywords) * 4;
    const tagHits = countKeywordHits(tags, cluster.keywords) * 4;
    const excerptHits = countKeywordHits(excerpt, cluster.keywords) * 3;
    const contentHits = countKeywordHits(content, cluster.keywords) * 1;

    return {
      clusterName: cluster.name,
      score: titleHits + slugHits + tagHits + excerptHits + contentHits
    };
  });

  return results.sort((a, b) => b.score - a.score);
}

import { INTERNAL_LINKING_CONFIG, TOOLS_HUB_URL } from "../../config/internalLinkingConfig";
import type { PostRecord, SelectedLink } from "./types";

export function selectLinksForPost(post: PostRecord, clusterName: string): SelectedLink[] {
  const cluster = INTERNAL_LINKING_CONFIG.find((c) => c.name === clusterName);
  if (!cluster) return [];

  const currentPostUrl = `/post/${post.slug}`;
  const selected: SelectedLink[] = [];

  selected.push({
    type: "primary-tool",
    title: cluster.primaryTool.title,
    url: cluster.primaryTool.url,
    keywords: cluster.primaryTool.keywords
  });

  for (const article of cluster.relatedArticles) {
    if (article.url === currentPostUrl) continue;

    selected.push({
      type: "related-article",
      title: article.title,
      url: article.url,
      keywords: article.keywords
    });

    if (selected.filter((x) => x.type === "related-article").length >= 2) break;
  }

  if (cluster.includeToolsHub) {
    selected.push({
      type: "tools-hub",
      title: "Apps & Tools",
      url: TOOLS_HUB_URL,
      keywords: ["tools", "apps & tools", "free tools"]
    });
  }

  return selected.slice(0, 4);
}

import type { InjectionRecord, SelectedLink } from "./types";

type InjectOptions = {
  maxLinks: number;
};

function escapeRegex(value: string): string {
  return value.replace(/[.*+?^${}()|[\]\\]/g, "\\$&");
}

function hasUrl(content: string, url: string): boolean {
  return content.includes(`href="${url}"`) || content.includes(`](${url})`);
}

function splitByProtectedBlocks(content: string): string[] {
  const protectedPattern =
    /(```[\s\S]*?```|`[^`]*`|<a\b[\s\S]*?<\/a>|!$begin:math:display$\[\^$end:math:display$]*\]$begin:math:text$\[\^\)\]\+$end:math:text$|^#{1,6}\s.*$)/gm;
  return content.split(protectedPattern);
}

function getProtectedMatches(content: string): string[] {
  const protectedPattern =
    /(```[\s\S]*?```|`[^`]*`|<a\b[\s\S]*?<\/a>|!\[[^\]]*\]\([^)]+\)|^#{1,6}\s.*$)/gm;
  return content.match(protectedPattern) || [];
}

function rebuildContent(unprotected: string[], protectedParts: string[]): string {
  let result = "";
  const maxLen = Math.max(unprotected.length, protectedParts.length);

  for (let i = 0; i < maxLen; i++) {
    if (i < unprotected.length) result += unprotected[i];
    if (i < protectedParts.length) result += protectedParts[i];
  }

  return result;
}

function injectMarkdownLink(
  text: string,
  link: SelectedLink
): { text: string; inserted: boolean; anchorText?: string } {
  for (const keyword of link.keywords) {
    const pattern = new RegExp(`\\b(${escapeRegex(keyword)})\\b`, "i");
    const match = text.match(pattern);
    if (!match) continue;

    const anchorText = match[1];

    const sentencePattern = new RegExp(
      `([^.!?\\n]*?)\\b${escapeRegex(anchorText)}\\b([^.!?\\n]*[.!?]?)`,
      "i"
    );
    const sentenceMatch = text.match(sentencePattern);

    if (sentenceMatch) {
      const fullSentence = sentenceMatch[0];
      const existingMarkdownLinksInSentence = (fullSentence.match(/\[[^\]]+\]\([^)]+\)/g) || []).length;
      const existingHtmlLinksInSentence = (fullSentence.match(/<a\b[\s\S]*?<\/a>/g) || []).length;

      if (existingMarkdownLinksInSentence + existingHtmlLinksInSentence > 0) {
        continue;
      }
    }

    const replacement = `[${anchorText}](${link.url})`;

    return {
      text: text.replace(pattern, replacement),
      inserted: true,
      anchorText
    };
  }

  return { text, inserted: false };
}

export function injectInternalLinks(
  originalContent: string,
  selectedLinks: SelectedLink[],
  options: InjectOptions = { maxLinks: 4 }
): { updatedContent: string; injections: InjectionRecord[] } {
  let content = originalContent;
  const injections: InjectionRecord[] = [];
  const usedTargets = new Set<string>();

  const protectedParts = getProtectedMatches(content);
  let unprotectedParts = splitByProtectedBlocks(content);

  let insertedCount = 0;

  for (const link of selectedLinks) {
    if (insertedCount >= options.maxLinks) {
      injections.push({
        type: link.type,
        title: link.title,
        url: link.url,
        anchorText: "",
        inserted: false,
        reason: "max_links_reached"
      });
      continue;
    }

    if (usedTargets.has(link.url)) {
      injections.push({
        type: link.type,
        title: link.title,
        url: link.url,
        anchorText: "",
        inserted: false,
        reason: "duplicate_target"
      });
      continue;
    }

    if (hasUrl(content, link.url)) {
      injections.push({
        type: link.type,
        title: link.title,
        url: link.url,
        anchorText: "",
        inserted: false,
        reason: "already_exists"
      });
      usedTargets.add(link.url);
      continue;
    }

    let linked = false;

    for (let i = 0; i < unprotectedParts.length; i++) {
      const part = unprotectedParts[i];
      const result = injectMarkdownLink(part, link);

      if (result.inserted) {
        unprotectedParts[i] = result.text;
        linked = true;
        insertedCount++;
        usedTargets.add(link.url);

        injections.push({
          type: link.type,
          title: link.title,
          url: link.url,
          anchorText: result.anchorText || "",
          inserted: true
        });

        break;
      }
    }

    if (!linked) {
      injections.push({
        type: link.type,
        title: link.title,
        url: link.url,
        anchorText: "",
        inserted: false,
        reason: "no_safe_anchor_found"
      });
    }

    content = rebuildContent(unprotectedParts, protectedParts);
  }

  return {
    updatedContent: content,
    injections
  };
}

import { db } from "../db";
import type { LinkerRunResult } from "./types";

export async function logInternalLinkerRun(result: LinkerRunResult): Promise<void> {
  await db.run(
    `
    INSERT INTO internal_linker_logs
    (post_id, post_title, cluster_name, cluster_score, mode, changed, injections_json, notes_json)
    VALUES (?, ?, ?, ?, ?, ?, ?, ?)
    `,
    [
      result.postId,
      result.postTitle,
      result.clusterName,
      result.clusterScore,
      result.mode,
      result.changed ? 1 : 0,
      JSON.stringify(result.injections),
      JSON.stringify(result.notes)
    ]
  );
}

import { db } from "../db";
import { scorePostAgainstClusters } from "./scorePost";
import { selectLinksForPost } from "./selectLinks";
import { injectInternalLinks } from "./injectLinks";
import { logInternalLinkerRun } from "./logger";
import type { LinkerRunResult, PostRecord } from "./types";

const MIN_CLUSTER_SCORE = 3;
const MIN_CONTENT_LENGTH = 300;

export async function getAllPublishedPosts(): Promise<PostRecord[]> {
  return db.all<PostRecord[]>(
    `
    SELECT id, title, slug, excerpt, content, tags, status
    FROM posts
    WHERE status = 'published'
    ORDER BY id DESC
    `
  );
}

export async function getPostById(postId: number): Promise<PostRecord | undefined> {
  return db.get<PostRecord>(
    `
    SELECT id, title, slug, excerpt, content, tags, status
    FROM posts
    WHERE id = ?
    `,
    [postId]
  );
}

export async function updatePostContent(postId: number, content: string): Promise<void> {
  await db.run(
    `
    UPDATE posts
    SET content = ?, updated_at = CURRENT_TIMESTAMP
    WHERE id = ?
    `,
    [content, postId]
  );
}

export async function runInternalLinkerForPost(
  post: PostRecord,
  mode: "dry-run" | "apply"
): Promise<LinkerRunResult> {
  const notes: string[] = [];

  if (!post.content || post.content.trim().length < MIN_CONTENT_LENGTH) {
    const result: LinkerRunResult = {
      postId: post.id,
      postTitle: post.title,
      clusterName: null,
      clusterScore: 0,
      mode,
      updatedContent: post.content,
      changed: false,
      injections: [],
      notes: ["post_too_short"]
    };

    await logInternalLinkerRun(result);
    return result;
  }

  const matches = scorePostAgainstClusters(post);
  const best = matches[0];

  if (!best || best.score < MIN_CLUSTER_SCORE) {
    const result: LinkerRunResult = {
      postId: post.id,
      postTitle: post.title,
      clusterName: null,
      clusterScore: best?.score || 0,
      mode,
      updatedContent: post.content,
      changed: false,
      injections: [],
      notes: ["no_cluster_confident_match"]
    };

    await logInternalLinkerRun(result);
    return result;
  }

  const selectedLinks = selectLinksForPost(post, best.clusterName);
  const { updatedContent, injections } = injectInternalLinks(post.content, selectedLinks, {
    maxLinks: 4
  });

  const changed = updatedContent !== post.content;

  if (!changed) {
    notes.push("no_changes_made");
  }

  if (mode === "apply" && changed) {
    await updatePostContent(post.id, updatedContent);
    notes.push("content_updated");
  }

  const result: LinkerRunResult = {
    postId: post.id,
    postTitle: post.title,
    clusterName: best.clusterName,
    clusterScore: best.score,
    mode,
    updatedContent,
    changed,
    injections,
    notes
  };

  await logInternalLinkerRun(result);
  return result;
}

export async function runInternalLinkerBatch(
  mode: "dry-run" | "apply",
  limit = 25
): Promise<LinkerRunResult[]> {
  const posts = await getAllPublishedPosts();
  const sliced = posts.slice(0, limit);

  const results: LinkerRunResult[] = [];

  for (const post of sliced) {
    const result = await runInternalLinkerForPost(post, mode);
    results.push(result);
  }

  return results;
}

import { Router } from "express";
import { db } from "../lib/db";
import {
  getPostById,
  runInternalLinkerBatch,
  runInternalLinkerForPost
} from "../lib/internalLinker/runInternalLinker";

const router = Router();

router.get("/admin/internal-linker", async (_req, res, next) => {
  try {
    const posts = await db.all(`
      SELECT id, title, slug, status, updated_at
      FROM posts
      ORDER BY id DESC
      LIMIT 50
    `);

    const logs = await db.all(`
      SELECT *
      FROM internal_linker_logs
      ORDER BY id DESC
      LIMIT 50
    `);

    res.render("admin/internal-linker", {
      posts,
      logs,
      results: []
    });
  } catch (err) {
    next(err);
  }
});

router.post("/admin/internal-linker/post/:id/dry-run", async (req, res, next) => {
  try {
    const postId = Number(req.params.id);
    const post = await getPostById(postId);

    if (!post) {
      return res.status(404).send("Post not found");
    }

    const result = await runInternalLinkerForPost(post, "dry-run");

    const posts = await db.all(`
      SELECT id, title, slug, status, updated_at
      FROM posts
      ORDER BY id DESC
      LIMIT 50
    `);

    const logs = await db.all(`
      SELECT *
      FROM internal_linker_logs
      ORDER BY id DESC
      LIMIT 50
    `);

    res.render("admin/internal-linker", {
      posts,
      logs,
      results: [result]
    });
  } catch (err) {
    next(err);
  }
});

router.post("/admin/internal-linker/post/:id/apply", async (req, res, next) => {
  try {
    const postId = Number(req.params.id);
    const post = await getPostById(postId);

    if (!post) {
      return res.status(404).send("Post not found");
    }

    const result = await runInternalLinkerForPost(post, "apply");

    const posts = await db.all(`
      SELECT id, title, slug, status, updated_at
      FROM posts
      ORDER BY id DESC
      LIMIT 50
    `);

    const logs = await db.all(`
      SELECT *
      FROM internal_linker_logs
      ORDER BY id DESC
      LIMIT 50
    `);

    res.render("admin/internal-linker", {
      posts,
      logs,
      results: [result]
    });
  } catch (err) {
    next(err);
  }
});

router.post("/admin/internal-linker/batch", async (req, res, next) => {
  try {
    const mode = req.body.mode === "apply" ? "apply" : "dry-run";
    const limit = Number(req.body.limit || 25);

    const results = await runInternalLinkerBatch(mode, limit);

    const posts = await db.all(`
      SELECT id, title, slug, status, updated_at
      FROM posts
      ORDER BY id DESC
      LIMIT 50
    `);

    const logs = await db.all(`
      SELECT *
      FROM internal_linker_logs
      ORDER BY id DESC
      LIMIT 50
    `);

    res.render("admin/internal-linker", {
      posts,
      logs,
      results
    });
  } catch (err) {
    next(err);
  }
});

export default router;
